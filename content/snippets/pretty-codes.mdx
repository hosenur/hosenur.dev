```css showLineNumbers title="syntax.css"
div[data-rehype-pretty-code-fragment] {
  @apply font-dank-mono;
}

div[data-rehype-pretty-code-fragment] pre {
  @apply px-0 my-2 overflow-x-auto leading-relaxed;
}
div[data-rehype-pretty-code-title] {
  @apply dark:text-white text-black font-dank-mono font-bold m-0;
}
div[data-rehype-pretty-code-fragment] .line--highlighted {
  background-color: #353e4b;
}
div[data-rehype-pretty-code-fragment] .line {
  @apply px-2 font-dank-mono;
}
div[data-rehype-pretty-code-fragment] code {
  @apply grid font-dank-mono;
}

code[data-line-numbers] {
  counter-reset: lineNumber;
}

code[data-line-numbers] .line::before {
  counter-increment: lineNumber;
  content: counter(lineNumber);
  @apply text-slate-500 inline-block text-right mr-2.5 w-4;
}
```
